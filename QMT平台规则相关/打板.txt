def integrated_zxg_reader(source_type, stock_source, file_path, C): 
	'''读取自选股'''
	if source_type == '排除' and len(file_path) == 0:
		A.lst_to_skip = []
		return
	if stock_source == '同花顺(远航)路径':
		tree = ET.parse(file_path)  
		root = tree.getroot()
		lst_stocks = []
		# 查找Block name为"打板"的内容
		for block in root.findall('Block'):
			if source_type == '监控':
				if block.get('name') == '打板':
					for security in block.findall('security'):
						# print(f"Market: {security.get('market')}, Code: {security.get('code')}")
						lst_stocks.append(security.get('code') + '.' + security.get('market')[1:3])
					# print(lst_stocks)
			else:
				if block.get('name') == '打板排除':
					for security in block.findall('security'):
						# print(f"Market: {security.get('market')}, Code: {security.get('code')}")
						lst_stocks.append(security.get('code') + '.' + security.get('market')[1:3])

	elif stock_source == '同花顺(免费)路径':
		with open(file_path, 'r', encoding='gbk') as f:
			lines = f.readlines()
			# 如果有空行，会认为股票代码是一个回车换行，所以删除掉这个回车换行
			lines = [x[1:] for x in lines if len(x) >= 6]
			# print(lines)
			data = [line.split("|") for line in lines]
			lst_codes = data[0]
			lst_stocks = []
			for code in lst_codes:
				if len(code) > 6:
					code = code[-6:]
					
				elif len(code) == 6:
					code = code
				elif len(code) < 6:
					continue
				lst_stocks.append(code)
			# print(lst_stocks)
			lst_stocks = [x + '.SH' if x.startswith('6') else x + '.SZ' for x in lst_stocks] # 整理股票代码
	
	elif stock_source == '通达信(普及)路径':
		with open(file_path, 'r', encoding='gbk') as f:
			lines = f.readlines()
			# 如果有空行，会认为股票代码是一个回车换行，所以删除掉这个回车换行
			lines = [x[1:] for x in lines if len(x) >= 6]
			lst_stocks = [line.split()[0] for line in lines]
			lst_stocks = [x + '.SH' if x.startswith('6') else x + '.SZ' for x in lst_stocks] # 整理股票代码
	
	elif stock_source == 'QMT(板块名)':
		lst_stocks = C.get_stock_list_in_sector(file_path)
	
	# lst_stocks = [x for x in lst_stocks if x[:1] not in ['9','8','4']] # 排除北交所股票
	lst_stocks = list(set(A.lst_hs_ashare)&(set(lst_stocks)))
	if source_type == '监控':
		A.lst_all = lst_stocks
		print(f"监控股票为{len(A.lst_all)}只")
	elif source_type == '排除':
		A.lst_to_skip = lst_stocks
		print(f"排除股票为{len(A.lst_to_skip)}只")
	# print(f"前5只为：{A.lst_all[:5]}")
	
	return 

def get_up_stop_price(C):
	time11 = time.time()
	# integrated_zxg_reader('监控', A.paras['stock_source'], A.paras['file_path'], C)
	# print('所需要监控的股票数为：%s只' % len(A.lst_all))
	
	if A.counter_for_get_upstopprice == 0: # 首次启动或重启动，则把所有板块内股票都查询一遍
		A.dict_up_stop_price = {}
		A.dict_floatVol = {}
		A.lst_ST = []
		for stock_code in A.lst_hs_ashare:
			abc = C.get_instrumentdetail(stock_code)
			up_stop_price = abc['UpStopPrice']
			floatVol = abc["FloatVolumn"]
			display_name = abc['InstrumentName']
			if type(display_name) != str:
				print(stock_code, '未取到涨停价, 跳过此股')
				A.lst_program_stocks.append(stock_code)
				continue
			if 'ST' in display_name:
				A.lst_ST.append(stock_code)
			# print(stock_code, type(up_stop_price) == float)
			if type(up_stop_price) != float:
				print(stock_code, '未取到涨停价, 跳过此股')
				A.lst_program_stocks.append(stock_code)
				continue
			if type(floatVol) not in [float, int]:
				print(stock_code, '未取到流通市值, 跳过此股')
				A.lst_program_stocks.append(stock_code)
				continue
			A.dict_up_stop_price[stock_code] = round(up_stop_price,2)
			A.dict_floatVol[stock_code] = floatVol
		A.lst_program_stocks = list(set(A.lst_program_stocks))
		time22 = time.time()
		consumed_time = time22 - time11
		A.counter_for_get_upstopprice += 1
		print('取涨停价耗时: %.2f秒' %consumed_time)
		print('自本次启动以来第%s次获取涨跌停价格' % A.counter_for_get_upstopprice)
			
	else:
		return

def over_time(order, now_time, now, order_time, seconds_threshold):
	if (order.m_strInsertTime <= '113000' and  now_time.replace(':', '') <= '113000') or (order.m_strInsertTime >= '130000' and  now_time.replace(':', '') >= '130000'):
		seconds = (now - order_time).seconds #当前日期时间-委托日期时间,获取秒数时间差
	elif order.m_strInsertTime <= '113000' and  now_time.replace(':', '') >= '130000':
		seconds = (now - order_time).seconds - 5400#当前日期时间-委托日期时间,获取秒数时间差
	# print(second)
	if seconds > seconds_threshold:
		return True
	else:
		return False

def over_time_to_explosion(stock_code, now_time, now, explosion_time):
	if (A.dict_explosion_time[stock_code][-6:] <= '113000' and  now_time.replace(':', '') <= '113000') or (A.dict_explosion_time[stock_code][-6:] >= '130000' and  now_time.replace(':', '') >= '130000'):
		seconds = (now - explosion_time).seconds #当前日期时间-委托日期时间,获取秒数时间差
	elif A.dict_explosion_time[stock_code][-6:] <= '113000' and  now_time.replace(':', '') >= '130000':
		seconds = (now - explosion_time).seconds - 5400#当前日期时间-委托日期时间,获取秒数时间差
	# print(second)
	if A.explosion_rt_and_seconds[2] < seconds < A.explosion_rt_and_seconds[3]:
		return True
	else:
		return False

			if A.full_tick[stock_code]['lastClose'] * A.dict_floatVol[stock_code] / 1e8 < A.min_market_value_threshold or A.full_tick[stock_code]['lastClose'] * A.dict_floatVol[stock_code] / 1e8 > A.max_market_value_threshold:
				continue
				
		
		if A.paras['open_vs_lastclose_judge']:
			# print(stock_code, round(A.full_tick[stock_code]['open'],2) / round(A.full_tick[stock_code]['lastClose'],2) - 1 , A.paras['open_vs_lastclose_rt'])
			if round(A.full_tick[stock_code]['open'],2) / round(A.full_tick[stock_code]['lastClose'],2) - 1 < A.open_vs_lastclose_range[0] \
				or round(A.full_tick[stock_code]['open'],2) / round(A.full_tick[stock_code]['lastClose'],2) - 1 > A.open_vs_lastclose_range[1]:
				continue
		
		if A.paras['exchange_judge'] == '换手率大于':
			# 成交量是手数，需要乘以100
			exchange_rt = A.full_tick[stock_code]['volume'] * 100 / A.dict_floatVol[stock_code]
			print(stock_code, exchange_rt)
			if exchange_rt <= A.paras['exchange_threshold']:
				continue
			else:
				pass
		
		if A.paras['exchange_judge'] == '换手率小于':
			exchange_rt = A.full_tick[stock_code]['volume'] * 100 / A.dict_floatVol[stock_code]
			print(stock_code, exchange_rt)
			if exchange_rt >= A.paras['exchange_threshold']:
				continue
			else:
				pass
		
		
		if A.oscillation_judge:
			if (A.full_tick[stock_code]['high'] - A.full_tick[stock_code]['low']) / A.full_tick[stock_code]['lastClose'] < A.oscillation_range[0] or (A.full_tick[stock_code]['high'] - A.full_tick[stock_code]['low']) / A.full_tick[stock_code]['lastClose'] > A.oscillation_range[1]:
				continue
			else:
				pass
		if A.buy_vs_traded_judge:
			# 只在涨停时进行判断
			# print('tt', stock_code, round(A.full_tick[stock_code]['lastPrice'], 2), A.dict_up_stop_price[stock_code])
			if round(A.full_tick[stock_code]['bidPrice'][0], 2) == A.dict_up_stop_price[stock_code]:
				# 封成比小于设定的值，则返回
				# print('hh', stock_code, A.full_tick[stock_code]['bidVol'][0],  A.full_tick[stock_code]['volume'])
				if A.full_tick[stock_code]['bidVol'][0] / A.full_tick[stock_code]['volume'] <= A.buy_vs_traded_threshold:
					continue
					# 大于,则不返回，可以进行下一步
				else:
					pass
			# 不涨停，则不进行此指标的判断
			else:
				pass
		
		if A.volume_judge:
			if A.full_tick[stock_code]['volume'] <= A.dealed_volume_threshold:
				# print(stock_code,'成交量条件未满足')
				continue
		if A.amount_judge:
			# print(stock_code, A.full_tick[stock_code]['amount'], A.dealed_amount_threshold, '成交金额条件未满足')
			if A.full_tick[stock_code]['amount'] <= A.dealed_amount_threshold:
				
				continue
		
		buy(C, stock_code)   #进行买入动作
		if A.paras['upstop_type'] in ['D1一字炸封', 'D2非一炸封', 'D3所有炸封']:
			if stock_code in A.lst_condition_0_met:
				A.lst_condition_0_met.remove(stock_code)
			
			
	# 整个for循环结束，做一次数据拷贝
	if A.paras['upstop_type'] in ['D1一字炸封', 'D2非一炸封', 'D3所有炸封']:
		# lst_in_explosion = list(set(A.lst_condition_0_met + A.lst_condition_1_met))
		# print('发生炸板的：', lst_in_explosion)
		A.dict_last_second_data = copy.deepcopy(A.full_tick)
		
	time22 = time.time()
	consumed_time = time22 - time11
	print('第%s轮监控，耗时: %.2f秒' % (A.counter_for_monitoring, consumed_time), '本轮监控了%s只股票' % n)
	A.fake_round += 1
	A.counter_for_monitoring += 1  #监控多少次计数
	return

def buy(C, stock_code):
	"""执行买入操作"""
	# print('buy', stock_code)
	amount_for_buying = min(A.available_cash, A.amount_for_buying)
	trade_price = round(A.dict_up_stop_price[stock_code],2)  # 涨停价下单
	vol = int(amount_for_buying / trade_price / 100) * 100 # 计算可买手数   买入数量 向下取整到100的整数倍
	
	if (stock_code[:] != '68' and vol >= 100) or (stock_code[:2] == '68' and vol >= 200):
		msg = f"以{trade_price}元，买入 {stock_code, vol}股"
		print(msg)
		passorder(A.buy_code, 1101, A.acct, stock_code, 11, trade_price, vol, '买入', 2, A.buy_remark, C)
		A.available_cash -= trade_price * vol
		A.lst_ordered_buy.append(stock_code)
		A.lst_ordered_buy = list(set(A.lst_ordered_buy))
	else:
		print(stock_code, '买入股数不足交易所规定的要求，跳过')
